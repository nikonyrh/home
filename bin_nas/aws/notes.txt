# Prepare and encrypt subsets
cd /storage/data
mkdir -p /rapid/data/backup/Kuvat_10
time ~/bin/aws/find_sha1sum.sh Kuvat/Upload_10 | tee /rapid/data/backup/Kuvat_10/Kuvat_10.sha1.txt
time ~/bin/aws/encrypt_subsets.sh password /rapid/data/backup Kuvat/Upload_10

# Split subsets
cd /rapid/data/backup
time ~/bin/aws/split.sh Kuvat_10
time ~/bin/aws/find_etag.sh Kuvat_10 | tee Kuvat_10/Kuvat_10.etag.txt
time sha1sum Kuvat_10/*.zip* | tee Kuvat_10/Kuvat_10.sha1.aes.txt

# Upload subsets
time ~/bin/aws/upload_aes_parts.sh Kuvat_10/*.aes_p*
